'number' + 3 + 3

// 'number' — рядок

// якщо є рядок, оператор + працює як склеювання

'number' + 3        // "number3"

"number3" + 3       // "number33"

Результат: "number33"

__________________

null + 3

// null при додаванні перетворюється на 0

null + 3            // 0 + 3 = 3

Результат: 3

__________________

5 && "qwerty"

// оператор && повертає перше false або останнє true

// 5 — true

// "qwerty" — true

Результат: "qwerty"
__________________

+'40' + +'2' + "hillel";

// +'40' перетворює рядок у число

+'40'  // 40

// +'2' перетворює рядок у число

+'2'   // 2

40 + 2  // 42

// є рядок + працює як склеювання

42 + "hillel"   // "42hillel"

5 && "qwerty"       // "qwerty"

Результат: "42hillel"

__________________

'10' - 5 === 6;

// оператор - завжди працює з числами

// '10'  -> 10

'10' - 5    // 5

5 === 6   // false

Результат: "false"

__________________

true + false

true + false   // 1 + 0 = 1

Результат:  1

'6' + 3 ** 0;

// оператор ** має вищий пріоритет

// є рядок + працює як склеювання

'6' + 1   // "61"

Результат: "61"

__________________

12 / '6'
